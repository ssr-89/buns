const accordOther = document.getElementById("accord");
accordOther.addEventListener("click", change);
function change(event) {
  const targ = event.target;
  if (targ.tagName !== "H3") return;
  if (targ.classList.contains("select-spoller")) {
    hideAll();
  } else {
    hideAll();
    targ.classList.add("select-spoller");
    showText(targ.nextElementSibling);
  }
}
function hideAll() {
  const accordTitle = accordOther.querySelectorAll(".accord__title");
  const accordContent = accordOther.querySelectorAll(".accord-content");
  for (let i = 0; i < accordTitle.length; i++) {
    accordTitle[i].classList.remove("select-spoller");
  }
  for (let i = 0; i < accordContent.length; i++) {
    accordContent[i].style.height = "0";
  }
}
function showText(textEl) {
  textEl.style.height = textEl.scrollHeight + "px";
}

// перетаскиваемый элемент
const item = document.querySelector(".item");

// ячейки, в одну из которых перетаскивается элемент
const placeholders = document.querySelectorAll(".placeholder");

item.addEventListener("dragstart", dragstart);
item.addEventListener("dragend", dragend);

for (const placeholder of placeholders) {
  // событие, когда элемент (item) находится над placeholder
  placeholder.addEventListener("dragover", dragover);

  // событие, когда элемент (item) заходит на территорию placeholder
  placeholder.addEventListener("dragenter", dragenter);

  // событие, когда элемент (item) выходит из территории placeholder
  placeholder.addEventListener("dragleave", dragleave);

  // событие, когда элемент (item) отпустили на территории placeholder
  placeholder.addEventListener("drop", dragdrop);
}

// event.target === this
function dragstart(event) {
  console.log("drag start", event.target);
  event.target.classList.add("hold");

  // пропадает из первой ячейки
  setTimeout(() => {
    event.target.classList.add("hide");
  }, 0);
}

function dragend(event) {
  console.log("drag end", event.target);

  // переписывает все классы
  event.target.className = "item";
}

function dragover(event) {
  // console.log("drag over");
  event.preventDefault();
}

function dragenter(event) {
  console.log("drag enter");
  event.target.classList.add("hovered");
}

function dragleave(event) {
  console.log("drag leave");
  event.target.classList.remove("hovered");
}

function dragdrop(event) {
  console.log("drag drop");
  event.target.classList.remove("hovered");
  event.target.append(item);
}

const fill = document.querySelector('.fill');

  this.classList.add('hold');
  setTimeout(() => {
    this.className = 'invisible';
  }, 0);

  this.classList = 'fill';

function dragEnter(e) {
  e.preventDefault();
  this.classList.add('hovered');
}

function dragLeave() {
  this.classList.remove('hovered');

  e.preventDefault();

  this.classList = 'empty';
  this.append(fill);

for (const empty of emptys) {

  empty.addEventListener('dragleave', dragLeave);
  empty.addEventListener('dragover', dragOver);
  empty.addEventListener('drop', drop);

  const headerMenu = document.getElementById("header-menu");
  const headerBurger = document.getElementById("header-burger");
  const headerMenuLinks = document.querySelectorAll(".header-menu-link");

  function openHeaderMenu() {
    if (headerBurger.classList.contains("open-menu")) {
      headerMenu.classList.add("open-menu");
    } else {
      headerMenu.classList.remove("open-menu");
    }
    headerMenuLinks.forEach(function (el) {
      el.addEventListener("click", function () {
        headerBurger.classList.remove("open-menu");
        headerMenu.classList.remove("open-menu");
      });
    });
    document.addEventListener("click", (element) => {
      if (!element.target.closest(".header-burger, .header-menu")) {
        headerBurger.classList.remove("open-menu");
        headerMenu.classList.remove("open-menu");
      }
    });
    headerBurger.addEventListener("click", () => {
      headerBurger.classList.toggle("open-menu");
      headerMenu.classList.toggle("open-menu");
    });
  }
  openHeaderMenu();
});

function bodyLock() {

  // В этой строке кода создается константа t, которая содержит значение разницы между шириной окна браузера 
  // и шириной обертки (wrapper) модального окна.
  const t = window.innerWidth - wrapper.offsetWidth + "px";

  // scrollY присваивает текущее значение вертикального смещения окна браузера (прокрутки) переменной e
  // и будет содержать числовое значение
  let e = window.scrollY;

  // Далее идёт условие, проверяющее, если в массиве lockPadding есть элементы.
  if (lockPadding.lenght > 0)

  // Если массив lockPadding содержит элементы, выполняется цикл перебора каждого элемента массива lockPadding.
  // В этом цикле добавляется стиль padding-right с значением t для каждого элемента.
    for (let e = 0; e < lockPadding.lenght; e++) {
      lockPadding[e].style.paddingRight = t;
    }
  // Здесь задается стиль padding-right для тела (body) с использованием значения t.
  (body.style.paddingRight = t),

  // Эта строка кода добавляет класс 'lock' к элементу тела (body),
  // что может использоваться для применения стилей или изменения поведения страницы.
    body.classList.add("lock"),

    // В атрибуте data-position для элемента тела (body) устанавливается значение e,
    // которое представляет текущее вертикальное смещение (прокрутку) окна браузера.
    (body.dataset.position = e),

    // Здесь устанавливается стиль top для элемента тела (body) с использованием отрицательного значения e,
    // чтобы сместить его вверх на величину текущей прокрутки страницы.
    (body.style.top = -e + "px");
}



function bodyUnLock() {

  // плавное отключение блокировки
  setTimeout(function () {

    // Создается переменная t, в которой сохраняется значение атрибута data-position тела (body).
    // Значение преобразуется в целое число с помощью parseInt().
    let t = parseInt(body.dataset.position, 10);

    // Здесь сбрасывается стиль top, чтобы вернуть его в исходное состояние.
    if (((body.style.top = "auto"), lockPadding.lenght > 0))

      // Если в массиве lockPadding есть элементы, происходит цикл,
      // который восстанавливает стиль padding-right для каждого элемента, устанавливая его значение равным "0px".
      for (let t = 0; t < lockPadding.lenght; t++) {
        lockPadding[t].style.paddingRight = "0px";
      }
    // сброс padding-right у body
    (body.style.paddingRight = "0px"),

      // удаление класса lock у body
      body.classList.remove("lock"),

      // Этот код выполняет прокрутку окна браузера к указанной вертикальной позиции t,
      // чтобы вернуть его в исходную позицию прокрутки.
      // Объект { top: t, left: 0 } передается в метод scroll объекта window для выполнения прокрутки.
      window.scroll({ top: t, left: 0 }),

      // удаление атрибута
      body.removeAttribute("data-position");
  }, timeout);
}

// функция определяет самое большое число id перебором массива,
// к которому функция прибавляет единицу и потом возвращает новый id
function itemId(arr) {
  let max = 0;
  for (let item of arr) {
    if (item.id > max) max = item.id;
  }
  return max + 1;
}
